// Основные настройки sf::Text

text.setFont(font);                         // задаёт шрифт, который будет использовать текст
text.setString("Text");                     // задаёт сам текст, который будет отображаться
text.setCharacterSize(24);                  // задаёт размер букв в пикселях
text.setFillColor(sf::Color::White);        // задаёт основной цвет текста
text.setPosition(20.0f, 20.0f);             // задаёт позицию текста на экране: X и Y

// Стили текста

text.setStyle(sf::Text::Bold);              // делает текст жирным
text.setStyle(sf::Text::Italic);            // делает текст курсивом
text.setStyle(sf::Text::Underlined);        // подчёркивает текст
text.setStyle(sf::Text::StrikeThrough);     // зачёркивает текст

// Обводка текста

text.setOutlineColor(sf::Color::Black);     // задаёт цвет обводки текста
text.setOutlineThickness(2.0f);             // задаёт толщину обводки текста

// Расстояния

text.setLetterSpacing(1.0f);                // задаёт расстояние между буквами
text.setLineSpacing(1.0f);                  // задаёт расстояние между строками

// Трансформация

text.setRotation(45.0f);                    // поворачивает текст на указанное количество градусов
text.setScale(1.0f, 1.0f);                  // задаёт масштаб текста по X и Y
text.setOrigin(0.0f, 0.0f);                 // задаёт точку отсчёта для позиции, поворота и масштаба
